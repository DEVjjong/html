전송기능 (form태그의 action에 지장된 서버페이지가 실행 되어진다.)
			- input type="submit"
			- button type="submit"
			- input type="image"
			- button
		
		일반버튼 (전송기능이 없다. script를 통해서 이벤트에 의한 기능을 부여한다.)
			- input type="button"
			- button type="button"
			
			<input type="button" value="확인" onclick="proc()">
			<button type="button" onclick="proc()">확인</button>
			
			<script>
			function proc(){
			idval = document.ff.id.value;
			passval = document.ff.pass.value;
			
			document.ff.action = "f1.jsp?id=" + idval + "&pass=" + passval;
			document.ff.submit();
		}
		</script>
